class Televisor():
    def __init__(self):
        self.apagado=True
        self.canal_actual=1
        self.volumen=10
    def Prender(self):
            if self.apagado:
                print('El televisor esta apagado: Presione ON')
    def Estado(self):
            return self.apagado
    def Menu(self):
             if self.apagado:
                 print('El televisor se ha prendido, CANAL TV-CABLE 1: Presione opcion del menu.')
    def Apagar(self):
             if self.apagado==True:
                 print('El televisor se ha apagado')

    def Subir_Canal(self):
            self.canal_actual=1
            self.canal_actual=self.canal_actual + 1
            print('Nuevo CANAL TV-CABLE', self.canal_actual)
    def Subir_Volumen(self):
            self.volumen=10
            self.volumen=self.volumen + 10
            print('Volumen aumento a:',self.volumen)
            Mas_Volumen=str(input('mas volumen?: si - no:  '))
            if Mas_Volumen=='si':
                self.volumen=self.volumen + 20
                print('Volumen Maximo: ',self.volumen)
